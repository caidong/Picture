CREATE TABLE iteach_logs.pair_teacher_client
(
    `clientID` String,

    `clientType` Int16,
    `areaCode` String,
    `areaName` String,
    `schoolCode` String,
    `schoolName` String,
    `classCode` String,
    `className` String,
    `eventSrc` String,
    `eventTime` DateTime,
    `CPU` String,
    `CPUByApp` String,
    `memory` String,
    `memoryByApp` String,
    `wifi` Int16,
    `errCode` Int16,
    `errMsg` String,
    `eventType` Int16,
    `eventMsg` String,
    `platForm` String,
    `networkType` String,
    `sysVersion` String,
    `screenResolution` String,
    `models` String,
    `manufacturer` String,
    `memoryTotal` String,
    `lessonID` String,
    `teacherID` String,
    `clientSendTime` String,
    `serverReceiveTime` String,
    `serverSendTime` String,
    `clientReceiveTime` String,
    `eventValue` String,
    `kafkaOffset` String,
    `kafkaPartition` String,
    `processTime` DateTime
)
ENGINE = MergeTree
PARTITION BY toYYYYMMDD(eventTime)
